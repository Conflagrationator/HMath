(!) :: (BitPack a, KnownNat (BitSize a), Enum i) => a -> i -> Bit #

Get the bit at the specified bit index.

NB: Bit indices are DESCENDING.

>>> pack (7 :: Unsigned 6)
00_0111
>>> (7 :: Unsigned 6) ! 1
1
>>> (7 :: Unsigned 6) ! 5
0
>>> (7 :: Unsigned 6) ! 6

split :: (BitPack a, BitSize a ~ (m + n), KnownNat n) => a -> (BitVector m, BitVector n) #

Split a value of a bit size m + n into a tuple of values with size m and size n.

>>> pack (7 :: Unsigned 6)
00_0111
>>> split (7 :: Unsigned 6) :: (BitVector 2, BitVector 4)
(00,0111)

msb :: (BitPack a, KnownNat (BitSize a)) => a -> Bit #

Get the most significant bit.

>>> pack (-4 :: Signed 6)
11_1100
>>> msb (-4 :: Signed 6)
1
>>> pack (4 :: Signed 6)
00_0100
>>> msb (4 :: Signed 6)
0
